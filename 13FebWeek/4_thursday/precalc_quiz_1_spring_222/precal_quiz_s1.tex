\question


\begin{multicols}{2}
 Which of the following represents the solution set for the  nonsquare system below?

$
\systeme{
2x - 3y + z = -2,
-4x + 9y + z = 7
}

\columnbreak
\begin{choices}
\choice $ x = -\frac{5}{6}a - \frac{1}{6} , y = \frac{7-a}{9}, z = a$
\choice $ x = \frac{1}{2} - 2a, y = 1 - a, z = a $
\choice $ x = \frac{1}{2} + 2a, y = 1 +a, z = a $
\choice This is an inconsistent system.
\end{choices}
\vspace{0.5cm}

\end{multicols}
